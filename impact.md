// Animate stat counters on scroll
document.addEventListener('DOMContentLoaded', function() {
  var counters = document.querySelectorAll('.stat-number');
  var speed = 60;

  var animateCounter = function(counter) {
    var target = +counter.getAttribute('data-target');
    var suffix = target >= 20 ? '+' : '';
    var increment = Math.max(1, Math.floor(target / speed));
    var current = 0;

    var updateCount = function() {
      current += increment;
      if (current >= target) {
        counter.innerText = target + suffix;
        return;
      }
      counter.innerText = current;
      requestAnimationFrame(updateCount);
    };
    updateCount();
  };

  var statObserver = new IntersectionObserver(function(entries) {
    entries.forEach(function(entry) {
      if (entry.isIntersecting) {
        animateCounter(entry.target);
        statObserver.unobserve(entry.target);
      }
    });
  }, { threshold: 0.5 });

  counters.forEach(function(counter) { statObserver.observe(counter); });

  // Timeline scroll animations
  var timelineEvents = document.querySelectorAll('.timeline-event');

  var timelineObserver = new IntersectionObserver(function(entries) {
    entries.forEach(function(entry) {
      if (entry.isIntersecting) {
        entry.target.classList.add('is-visible');
        timelineObserver.unobserve(entry.target);
      }
    });
  }, { threshold: 0.15, rootMargin: '0px 0px -50px 0px' });

  timelineEvents.forEach(function(event) { timelineObserver.observe(event); });
});
</script>
